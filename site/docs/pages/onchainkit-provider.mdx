# `<OnchainKitProvider />`

Provides the OnchainKit React Context to the app.

## Usage

```tsx [code]
import { base } from 'viem/chains'; // [!code focus]
import { OnchainKitProvider } from '@coinbase/onchainkit'; // [!code focus]

const App = () => {
  return (
    <OnchainKitProvider chain={base}>
      {' '}
      // [!code focus]
      <MyComponent />
    </OnchainKitProvider> // [!code focus]
  );
};
```
